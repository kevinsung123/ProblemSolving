    void list::insertF(int val) {
      node* temp=new node;//새로운 노드를 할당
      temp->ele = val;//노드에 값을 넣는다. 
      if (head == NULL) {//리스트에 아무것도 없을 때
        head=temp;//head에 temp를 넣는다.
        temp->next = NULL;
        tail = head;
      }
      else {
        temp->next = head;//temp와 원래 head를 연결
        head = temp;//head를 temp로
      }
    }
    void list::insertL(int val) {
      node* temp=new node;//새로운 노드를 할당

      }

        current = head;

        temp->next = NULL;
         tail=temp;//temp가 tail
      }
    }
    void list::deleteF() {
      
      if (head == NULL) {//노드가 없을 때
        cout << "노드가 없습니다."<<endl;

int Partition(int n);
int main(void)
{
	int T;
	int * n;
	cin >> T;
	n = new int[T];
	for (int i = 0; i < T; i++)
		cin >> n[i];
	for (int i = 0; i < T; i++)
		cout << Partition(n[i]) << endl;
}
int Partition(int n){
	if (n == 0) return 0;
	if (n == 1) return 1;
	if (n == 2) return 2;
	if (n == 3) return 4;
	return Partition(n - 1) + Partition(n - 2) + Partition(n - 3);
}
